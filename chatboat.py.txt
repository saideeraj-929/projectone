import urllib.request
import json
from datetime import datetime

# Function to get weather using Weatherstack
def get_weather(city):
    api_key = "YOUR_WEATHERSTACK_KEY"  # Replace with your Weatherstack API key
    url = f"http://api.weatherstack.com/current?access_key={api_key}&query={city}"
    try:
        with urllib.request.urlopen(url) as response:
            data = json.loads(response.read().decode())
            print("DEBUG: Raw API response:", data)  # helpful for troubleshooting
            
            if "current" in data:
                temperature = data["current"]["temperature"]
                weather_desc = data["current"]["weather_descriptions"][0]
                return f"Temperature: {temperature}°C\nCondition: {weather_desc}"
            else:
                return f"Error: {data.get('error', {}).get('info', 'Unable to fetch weather')}"
    except Exception as e:
        return f"Error fetching weather: {e}"
