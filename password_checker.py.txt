import re

def check_password_strength(password):
    strength=0
    suggestion=[]
    
    if len(password)>=8:
        strength+=1
    else:
        suggestion.append("use at least 8 charecters.")
    if re.search(r"[A-Z]",password):
        strength+=1
    else:
        suggestion.append("Add upper case letters.")
    if re.search(r"[a-z]",password):
        strength+=1
    else:
        suggestion.append("Add lower case letters.")
    if re.search(r"[0-9]",password):
        strength+=1
    else:
        suggestion.append("Invalid numbers.")
    if re.search(r"[!@#$%^&*(),.?/:{}|<>]",password):
        strength+=1
    else:
        suggestion.append("Include special charecters.")
    if strength <=2:
        rating="weak"
    elif strength==3 or strength==4:
        rating="Medium"
    else:
        rating="Strong"
    return rating,suggestion
    
if __name__=="__main__":
        user_password=input("Enter a password to check:")
        rating,tips = check_password_strength(user_password)
        print("Password Strength",{rating})
        if tips:
            print("Suggestions:")
            for tips in tips:
                print("-",tips)
